46/*!
47 * @note This is a user type function.
48 * Note that there is predeclred device structure "BMM350"
49 * You must call this function before getting the data.
50 * @brief This function to initialize the BMM350 registers and enable Normal Mode.
51 *
52 * @param[out] dev structure pointer to device information structure.
53 * @param[in] hi2c structure pointer to the HAL I2C.
54 *
55 * @return bmm350_status_e -> enum that holds the status of the initialization process.
56 * @code
57 * bmm350_init(&BMM350, &hi2c2);
58 * @endcode
59 */
60bmm350_status_e bmm350_init(struct bmm350_dev_s *dev, I2C_HandleTypeDef *hi2c){
61 bmm350_status_e rslt;
62
63 rslt = bmm350_init_struct(dev, hi2c);
64 // Chip ID
65 uint8_t chip_id;
66
67 // Variable to store the command to power-off the OTP
68 uint8_t otp_cmd = BMM350_OTP_CMD_PWR_OFF_OTP;
69
70 /* Variable to store soft-reset command */
71 uint8_t soft_reset;
72
73 if(rslt == BMM350_OK){
74 rslt = check_device(dev);
75 if(rslt == BMM350_OK){
76 rslt = null_ptr_check(dev);
77 }
78 }
79 if(rslt == BMM350_OK){
80 dev->chip_id = 0;
81
82 // Enable all axis
83 dev->axis_en = 0x07;
84
85 dev->delay_us(BMM350_START_UP_TIME_FROM_POR);
86
87 if(rslt == BMM350_OK){
88 /* Soft-reset */
89 soft_reset = BMM350_CMD_SOFTRESET;
90
91 /* Set the command in the command register */
92 rslt = dev->write(dev, BMM350_REG_CMD, &soft_reset, 1);
93 if(rslt == BMM350_OK){
94 dev->delay_us(BMM350_SOFT_RESET_DELAY);
95 }
96 }
97
98 if(rslt == BMM350_OK){
99 rslt = dev->read(dev, BMM350_REG_CHIP_ID, &chip_id, 1);
100 if(rslt == BMM350_OK){
101 dev->chip_id = chip_id;
102 }
103 }
104
105 /* Check for chip id validity */
106 if((rslt == BMM350_OK) && (dev->chip_id == BMM350_CHIP_ID)){
107 /* Download OTP memory */
108 rslt = otp_dump_after_boot(dev);
109 if(rslt == BMM350_OK){
110 /* Power off OTP */
111 rslt = dev->write(dev, BMM350_REG_OTP_CMD_REG, &otp_cmd, 1);
112 dev->delay_us(BMM350_SOFT_RESET_DELAY);
113 if(rslt == BMM350_OK){
114 rslt = bmm350_magnetic_reset_and_wait(dev);
115 dev->delay_us(BMM350_GOTO_SUSPEND_DELAY);
116 }
117 }
118 }else{
119 rslt = BMM350_E_DEV_NOT_FOUND;
120 }
121
122 if(rslt == BMM350_OK){
123 rslt = interrupt_config(dev);
124 }
125
126 if(rslt == BMM350_OK){
127 rslt = bmm350_odr_avg_config(dev, BMM350_ODR_400HZ, BMM350_AVG_NO_AVG);
128 if((rslt == BMM350_OK)){
129 rslt = bmm350_set_powermode(dev, BMM350_NORMAL_MODE);
130 }
131 }
132
133 if(rslt == BMM350_OK){
134 rslt = bmm350_get_pmu_cmd_status(dev, &dev->pmu_cmd_status);
135 }
136 }
137 return rslt;
138}

140/*!
141 * @note This is a user type function.
142 * @brief This function is used to get yaw value
143 *
144 * @param[in] dev structure pointer to device information structure
145 * @param[in] unit enum to specify the type of the returned value (radian or degree)
146 * @param[in] range enum to specify the range of degree returned **if unit is radian then put NON**
147 *
148 * @retval yaw -> The YAW angle value
149 * @code
150 * yaw = BMM350_read_yaw(&BMM350, DEGREE, FROM_0_TO_360);
151 * @endcode
152 */
153float bmm350_read_yaw(struct bmm350_dev_s *dev, bmm350_ang_unt_rng_e unit, bmm350_ang_unt_rng_e range){
154 bmm350_status_e rslt;
155 static float yaw[2] = {0.0f}; // [0] radian, [1] degree
156 static float NUV[2] = {0.0f}; // Normalized unit vector for, [0] X, [1] Y
157
158 rslt = read_intr(dev, &dev->data);
159 if(rslt == BMM350_INT_DATA_READY){
160 NUV[0] = (dev->data.x / sqrt((pow(dev->data.y, 2) + pow(dev->data.x, 2))));
161 NUV[1] = (dev->data.y / sqrt((pow(dev->data.y, 2) + pow(dev->data.x, 2))));
162
163 yaw[0] = atan2(-1 * NUV[1], NUV[0]);
164 dev->yaw[0] = (yaw[0] * (180.0 / PI)); // Initialize the raw yaw in the structure for kalman filtering
165 if(dev->yaw[0] < 0) dev->yaw[0] += 360.0; // Makes sure it is in the of 0 to 360 for better filtering
166
167 dev->yaw[1] = adaptive_filter(&BMM350_yaw_f, dev->yaw[0]);
168
169 yaw[1] = dev->yaw[1];
170
171 if(range == FROM_NEG_180_TO_POS_180){
172 yaw[1] -= (yaw[1] > 180)? 360.0f : 0;
173 }
174 }
175 return (unit == RADIAN)? yaw[0] : yaw[1];
176}

213 *
214 * @brief Performs a magnetic reset of the BMM350 sensor. This is necessary when:
215 * The sensor may have been exposed to excessively strong magnetic fields (e.g. >400 mT).
216 * Long periods of inactivity have caused capacitor discharge on the CRST pin.
217 * You're resuming from suspend mode and want to guarantee accuracy.
218 *
219 * @param[out] dev structure pointer to device information structure.

221 * @retval BMM350_OK -> as it says

226 * @endcode

229bmm350_status_e bmm350_magnetic_reset_and_wait(struct bmm350_dev_s *dev){
230 bmm350_status_e rslt;
231
232 uint8_t pmu_cmd = 0;
233 bmm350_pmu_cmd_status_s pmu_cmd_stat = { 0 };
234 uint8_t restore_normal = BMM350_DISABLE;
235
236 rslt = null_ptr_check(dev);
237
238// if ((rslt == BMM350_OK) && (dev->mraw_override) && (dev->var_id >= BMM350_MIN_VAR)){
239//
240// rslt = dev->mraw_override(dev);
241// }else{
242 /* Read PMU CMD status */
243 rslt = bmm350_get_pmu_cmd_status(dev, &pmu_cmd_stat); // How Normal is set????
244
245 /* Check the powermode is normal before performing magnetic reset */
246 if((rslt == BMM350_OK) && (pmu_cmd_stat.status_0.pwr_mode_is_normal == BMM350_ENABLE)){
247 restore_normal = BMM350_ENABLE;
248
249 /* Reset can only can be triggered in suspend */
250 rslt = bmm350_set_powermode(dev, BMM350_SUSPEND_MODE);
251 }
252
253 if(rslt == BMM350_OK){
254 /* Set BR to PMU_CMD register */
255 pmu_cmd = BMM350_PMU_CMD_BR;
256
257 rslt = dev->write(dev, BMM350_REG_PMU_CMD, &pmu_cmd, 1);
258 if(rslt == BMM350_OK){
259 dev->delay_us(BMM350_BR_DELAY);
260 }
261 }
262
263 if(rslt == BMM350_OK){
264 /* Verify if PMU_CMD_STATUS_0 register has BR set */
265 rslt = bmm350_get_pmu_cmd_status(dev, &pmu_cmd_stat);
266
267 if ((rslt == BMM350_OK) && (pmu_cmd_stat.status_0.pmu_cmd_value != BMM350_PMU_CMD_STATUS_0_BR))
268 {
269 rslt = BMM350_E_PMU_CMD_VALUE;
270 }
271 }
272
273 if(rslt == BMM350_OK){
274 /* Set FGR to PMU_CMD register */
275 pmu_cmd = BMM350_PMU_CMD_FGR;
276
277 rslt = dev->write(dev, BMM350_REG_PMU_CMD, &pmu_cmd, 1);
278 if(rslt == BMM350_OK){
279 dev->delay_us(BMM350_FGR_DELAY);
280 }
281 }
282
283 if(rslt == BMM350_OK){
284 /* Verify if PMU_CMD_STATUS_0 register has FGR set */
285 rslt = bmm350_get_pmu_cmd_status(dev, &pmu_cmd_stat);
286
287 if ((rslt == BMM350_OK) && (pmu_cmd_stat.status_0.pmu_cmd_value != BMM350_PMU_CMD_STATUS_0_FGR))
288 {
289 rslt = BMM350_E_PMU_CMD_VALUE;
290 }
291 }
292
293 if ((rslt == BMM350_OK) && (restore_normal == BMM350_ENABLE))
294 {
295 rslt = bmm350_set_powermode(dev, BMM350_NORMAL_MODE);
296 }
297
298// }
299
300 return rslt;
301}

312/*!
313 * @note This is a internal & user type function
314
315 * @brief This function filters the data of angle got from magnetic readings
316 *
317 * @param[in] filter: structure pointer to filter
318 * @param[out] raw_data: The angle data
319 * @return filtered angle
320 *
321 * @code
322 * adaptive_filter(&BMM350_yaw_f, dev->yaw[0]);
323 * @endcode
324 */
325
326float adaptive_filter(bmm350_filter *filter, float raw_data){
327 float unwrapped = unwrap_angle(filter->data, raw_data);
328 float delta = fabsf(unwrapped - filter->data);
329
330
331 filter->alpha = ((delta > CHANGE_THRESHOLD)? ALPHA_MAX : ALPHA_MIN /*+ (delta / CHANGE_THRESHOLD) * (ALPHA_MAX - ALPHA_MIN)*/);
332
333 filter->data = filter->alpha * unwrapped + (1.0f - filter->alpha) * filter->data;
334 return wrap_360(filter->data);
335}

475/*!
476 * @note This is an internal & user type function
477 * @brief This internal function is used to perform compensation calculations
478 * for raw magnetometer and temperature data, and assign it to structure.
479 *
480 * @param[in] dev structure pointer to device information structure.
481 * @param[out] mag_temp_data structure pointer to the magnetic field and temperature
482 * data x, y, z, temperature.
483 *
484 * @retval BMM350_OK -> as it says
485 * @retval BMM350_E_NULL_PTR -> Second argument is NULL
486 *
487 * @code
488 * bmm350_get_compensated_mag_xyz_temp_data(&BMM350, &raw_data);
489 * @endcode
490 */
491
492bmm350_status_e bmm350_get_compensated_mag_xyz_temp_data(struct bmm350_dev_s *dev, bmm350_mag_temp_data_s *mag_temp_data){
493 bmm350_status_e rslt;
494
495 uint8_t indx;
496 float out_data[4] = { 0.0f };
497 float dut_offset_coef[3], dut_sensit_coef[3], dut_tco[3], dut_tcs[3];
498 float cr_ax_comp_x, cr_ax_comp_y, cr_ax_comp_z;
499
500 if(mag_temp_data != NULL){
501
502 rslt = read_out_raw_data(dev, out_data);
503 if(rslt == BMM350_OK){
504 /* Apply compensation to temperature reading */
505 out_data[3] = (1 + dev->mag_comp.dut_sensit_coef.t_sens) * out_data[3] + dev->mag_comp.dut_offset_coef.t_offs;
506
507 /* Store magnetic compensation structure to an array */
508 dut_offset_coef[0] = dev->mag_comp.dut_offset_coef.offset_x;
509 dut_offset_coef[1] = dev->mag_comp.dut_offset_coef.offset_y;
510 dut_offset_coef[2] = dev->mag_comp.dut_offset_coef.offset_z;
511
512 dut_sensit_coef[0] = dev->mag_comp.dut_sensit_coef.sens_x;
513 dut_sensit_coef[1] = dev->mag_comp.dut_sensit_coef.sens_y;
514 dut_sensit_coef[2] = dev->mag_comp.dut_sensit_coef.sens_z;
515
516 dut_tco[0] = dev->mag_comp.dut_tco.tco_x;
517 dut_tco[1] = dev->mag_comp.dut_tco.tco_y;
518 dut_tco[2] = dev->mag_comp.dut_tco.tco_z;
519
520 dut_tcs[0] = dev->mag_comp.dut_tcs.tcs_x;
521 dut_tcs[1] = dev->mag_comp.dut_tcs.tcs_y;
522 dut_tcs[2] = dev->mag_comp.dut_tcs.tcs_z;
523
524 /* Compensate raw magnetic data */
525 for (indx = 0; indx < 3; indx++)
526 {
527 out_data[indx] *= 1 + dut_sensit_coef[indx];
528 out_data[indx] += dut_offset_coef[indx];
529 out_data[indx] += dut_tco[indx] * (out_data[3] - dev->mag_comp.dut_t0);
530 out_data[indx] /= 1 + dut_tcs[indx] * (out_data[3] - dev->mag_comp.dut_t0);
531 }
532
533 cr_ax_comp_x = (out_data[0] - dev->mag_comp.cross_axis.cross_x_y * out_data[1]) /
534 (1 - dev->mag_comp.cross_axis.cross_y_x * dev->mag_comp.cross_axis.cross_x_y);
535 cr_ax_comp_y = (out_data[1] - dev->mag_comp.cross_axis.cross_y_x * out_data[0]) /
536 (1 - dev->mag_comp.cross_axis.cross_y_x * dev->mag_comp.cross_axis.cross_x_y);
537 cr_ax_comp_z =
538 (out_data[2] +
539 (out_data[0] *
540 (dev->mag_comp.cross_axis.cross_y_x * dev->mag_comp.cross_axis.cross_z_y -
541 dev->mag_comp.cross_axis.cross_z_x) - out_data[1] *
542 (dev->mag_comp.cross_axis.cross_z_y - dev->mag_comp.cross_axis.cross_x_y *
543 dev->mag_comp.cross_axis.cross_z_x)) /
544 (1 - dev->mag_comp.cross_axis.cross_y_x * dev->mag_comp.cross_axis.cross_x_y));
545
546 out_data[0] = cr_ax_comp_x;
547 out_data[1] = cr_ax_comp_y;
548 out_data[2] = cr_ax_comp_z;
549 }
550
551 if (rslt == BMM350_OK)
552 {
553 if ((dev->axis_en & BMM350_EN_X_MSK) == BMM350_DISABLE)
554 {
555 mag_temp_data->x = BMM350_DISABLE;
556 }
557 else
558 {
559 mag_temp_data->x = out_data[0];
560 }
561
562 if ((dev->axis_en & BMM350_EN_Y_MSK) == BMM350_DISABLE)
563 {
564 mag_temp_data->y = BMM350_DISABLE;
565 }
566 else
567 {
568 mag_temp_data->y = out_data[1];
569 }
570
571 if ((dev->axis_en & BMM350_EN_Z_MSK) == BMM350_DISABLE)
572 {
573 mag_temp_data->z = BMM350_DISABLE;
574 }
575 else
576 {
577 mag_temp_data->z = out_data[2];
578 }
579
580 mag_temp_data->temperature = out_data[3];
581 }
582 }else{
583 rslt = BMM350_E_NULL_PTR;
584 }
585 return rslt;
586}

588/*!

591 *

593 * @param[in] powermode: enum from bmm350_power_modes_e for indicating the power mode
594 *

602
603bmm350_status_e bmm350_set_powermode(struct bmm350_dev_s *dev, bmm350_power_modes_e powermode){
604 bmm350_status_e rslt;
605
606 uint8_t last_pwr_mode;
607 uint8_t reg_data;
608
609 /* Check for null pointer in the device structure */
610 rslt = null_ptr_check(dev);
611 if(rslt == BMM350_OK){

1121 break;
1122 }

1124 dev->delay_us(delay_us);
1125 }
1126

1334int32_t fix_sign(uint32_t inval, int8_t number_of_bits){
